package com.leijia;

import java.util.Scanner;
public class leijia {
	  
	public static void main(String[] args) { 
	System.out.println("Please input a number:"); 
	Scanner s = new Scanner(System.in); 
	int a = s.nextInt(); //接收一个整数 
	int sum=0;
	for(int i=1; i<=a; i++)
	{
		sum=i+sum;
	}

	System.out.println("The input number is "+sum); 
	} 
}
